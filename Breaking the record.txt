
#include <bits/stdc++.h>

using namespace std;

int main()
{
    int i,n,max,min,x=1,c=0,t=0;
    cin>>n;
    int a[n];
    for(i=0;i<n;i++)
    {
        cin>>a[i];
    }
    max=a[0];
    min=a[0];
    for(i=1;i<n;i++)
    {
        if(a[i]>max)
        {
            max=a[i];
            c++;
        }
        if(a[i]<min)
        {
            min=a[i];
            t++;
        }
    }

    cout<<c<<" "<<t;
    return 0;
}
